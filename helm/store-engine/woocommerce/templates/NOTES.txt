# Check pod status

# View WordPress logs
kubectl logs -n {{ .Release.Namespace }} deployment/{{ include "store-instance.wordpress.fullname" . }} -f

# View MySQL logs
kubectl logs -n {{ .Release.Namespace }} statefulset/{{ include "store-instance.mysql.fullname" . }} -f

# Get admin password
kubectl get secret -n {{ .Release.Namespace }} {{ include "store-instance.wordpress.fullname" . }}-credentials -o jsonpath='{.data.admin-password}' | base64 -d
